#### Basic Usage

Write markdown here.

```tsx
import React from 'react';
import ReactDOM from 'react-dom';
import { Text } from 'demo';

ReactDOM.render(
  <div>
    <Text size="large" />
  </div>,
  mountNode,
);
```

```tsx
import React from 'react';
import { Text } from 'demo';

function App() {
  return (
    <div>
      <Text size="medium" />
    </div>
  );
}
export default App;
```
